    Write-Host ""
    Write-Host "${ANSI.PURPLE}${ANSI.BOLD}TASK HISTORY${ANSI.RESET}"
    Write-Host "${ANSI.BORDER}â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€${ANSI.RESET}"

        return
    }
    
    try {
        $history = Get-WinEvent -FilterHashtable @{LogName='Microsoft-Windows-TaskScheduler/Operational'; ID=200,201} | 
                   Where-Object { $_.Message -like "*$taskName*" } |
                   Sort-Object TimeCreated -Descending |
                   Select-Object -First 20
        
        if ($history.Count -eq 0) {
            Write-Host "${ANSI.TEXT_SECONDARY}No history found for task '$taskName'${ANSI.RESET}"
            return
        }
        
        Write-Host ""
        Write-Host "${ANSI.PURPLE}${ANSI.BOLD}RECENT HISTORY FOR '$taskName'${ANSI.RESET}"
        Write-Host "${ANSI.BORDER}â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€${ANSI.RESET}"
        
        $history | ForEach-Object {
            $time = $_.TimeCreated.ToString("yyyy-MM-dd HH:mm")
            $message = if ($_.Message.Length -gt 60) { $_.Message.Substring(0, 57) + "..." } else { $_.Message }
            
            Write-Host " ${ANSI.TEXT_SECONDARY}$time${ANSI.RESET}"
            Write-Host " ${ANSI.TEXT_PRIMARY}$message${ANSI.RESET}"
            Write-Host ""
        }
    } catch {
        Write-Host "${ANSI.RED}Ã— Error retrieving task history: $($_.Exception.Message)${ANSI.RESET}"
        Write-Host "${ANSI.TEXT_SECONDARY}Task Scheduler event logs may not be enabled${ANSI.RESET}"
    }
}

function Receive-JobOutput {

    }
    
    Write-Host ""

    if ($jobId -gt 0) {
        $selectedJob = $jobs[$jobId - 1]
        
        Write-Host ""
        Write-Host "${ANSI.TEXT_SECONDARY}Retrieving output for job '$($selectedJob.Name)'...${ANSI.RESET}"
        
        try {
            $output = Receive-Job -Id $selectedJob.Id -Keep
            if ($output) {
                Write-Host ""
                Write-Host "${ANSI.PURPLE}${ANSI.BOLD}JOB OUTPUT${ANSI.RESET}"
                Write-Host "${ANSI.BORDER}â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€${ANSI.RESET}"

            } else {
                Write-Host "${ANSI.TEXT_SECONDARY}No output available${ANSI.RESET}"
            }
        } catch {
            Write-Host "${ANSI.RED}Ã— Error receiving job output: $($_.Exception.Message)${ANSI.RESET}"
        }
    }
}
