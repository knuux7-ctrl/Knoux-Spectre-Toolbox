function Start-ServiceInteractive {
    Write-Host ""
    Write-Host "${ANSI.TEXT_SECONDARY}Enter service name to start:${ANSI.RESET}"
    Write-Host "${ANSI.PURPLE}>>${ANSI.RESET} " -NoNewline
    $serviceName = Read-Host

        Write-Host "${ANSI.RED}Ã— Service name cannot be empty${ANSI.RESET}"
        return

    try {
        $service = Get-Service -Name $serviceName -ErrorAction Stop

            Write-Host "${ANSI.TEXT_SECONDARY}Service '$serviceName' is already running${ANSI.RESET}"
            return

        Write-Host ""
        Write-Host "${ANSI.TEXT_SECONDARY}Starting service '$serviceName'...${ANSI.RESET}" -NoNewline
        
        if (-not (Test-AdminPrivilege)) {
            Write-Host ""
            Write-Host "${ANSI.ORANGE}âš  Administrator privileges required to start services${ANSI.RESET}"
            return
        }
        
        Start-Service -Name $serviceName

        Write-Host "${ANSI.RED}Ã— Service '$serviceName' not found${ANSI.RESET}"
