function Pull-DockerImage {
    Write-Host ""
    Write-Host "${ANSI.TEXT_SECONDARY}Enter image name to pull (e.g., nginx, redis):${ANSI.RESET}"
    Write-Host "${ANSI.PURPLE}>>${ANSI.RESET} " -NoNewline
    $image = Read-Host
    
    if ([string]::IsNullOrWhiteSpace($image)) {
        Write-Host "${ANSI.RED}Ã— Image name cannot be empty${ANSI.RESET}"
        return
    }
    
    Write-Host ""
    Write-Host "${ANSI.TEXT_SECONDARY}Pulling image '$image'...${ANSI.RESET}"
    Write-Host "${ANSI.TEXT_SECONDARY}This may take a few minutes depending on image size${ANSI.RESET}"
    Write-Host ""
    
    try {
        $process = Start-Process -FilePath "docker" -ArgumentList "pull", $image -NoNewWindow -PassThru -RedirectStandardOutput "$env:TEMP\docker_pull.out" -RedirectStandardError "$env:TEMP\docker_pull.err"
        
        # Show progress indicator
        Write-Host "${ANSI.TEXT_SECONDARY}Progress:${ANSI.RESET}" -NoNewline
        while (!$process.HasExited) {
            Write-Host "." -NoNewline
            Start-Sleep -Milliseconds 500
        }
        
        Write-Host ""
        
        if ($process.ExitCode -eq 0) {
            Write-Host "${ANSI.GREEN}âœ“ Image pulled successfully${ANSI.RESET}"
        } else {
            Write-Host "${ANSI.RED}Ã— Failed to pull image${ANSI.RESET}"
            $errorOutput = Get-Content "$env:TEMP\docker_pull.err" -ErrorAction SilentlyContinue
            if ($errorOutput) {
                Write-Host "${ANSI.TEXT_SECONDARY}Error: $errorOutput${ANSI.RESET}"
            }
        }
        
        # Clean up temp files
        Remove-Item "$env:TEMP\docker_pull.out" -ErrorAction SilentlyContinue
        Remove-Item "$env:TEMP\docker_pull.err" -ErrorAction SilentlyContinue
    } catch {
        Write-Host "${ANSI.RED}Ã— Error pulling image: $($_.Exception.Message)${ANSI.RESET}"
    }
}

function Run-DockerContainer {
    Write-Host ""
    Write-Host "${ANSI.TEXT_SECONDARY}Enter image name to run:${ANSI.RESET}"
    Write-Host "${ANSI.PURPLE}>>${ANSI.RESET} " -NoNewline
    $image = Read-Host
    
    if ([string]::IsNullOrWhiteSpace($image)) {
        Write-Host "${ANSI.RED}Ã— Image name cannot be empty${ANSI.RESET}"
        return
    }
    
    Write-Host ""
    Write-Host "${ANSI.TEXT_SECONDARY}Enter container name (optional):${ANSI.RESET}"
    Write-Host "${ANSI.PURPLE}>>${ANSI.RESET} " -NoNewline
    $name = Read-Host
    
    Write-Host ""
    Write-Host "${ANSI.TEXT_SECONDARY}Enter port mapping (optional, e.g., 8080:80):${ANSI.RESET}"
    Write-Host "${ANSI.PURPLE}>>${ANSI.RESET} " -NoNewline
    $portMapping = Read-Host
    
    Write-Host ""
    Write-Host "${ANSI.TEXT_SECONDARY}Run in detached mode?${ANSI.RESET}"
    $detached = Confirm-KnouxAction "Run detached?" "Y"
    
    $args = @("run")
    
    if ($detached) {
        $args += "-d"
    }
    
    if (-not [string]::IsNullOrWhiteSpace($name)) {
        $args += "--name"
        $args += $name
    }
    
    if (-not [string]::IsNullOrWhiteSpace($portMapping)) {
        $args += "-p"
        $args += $portMapping
    }
    
    $args += $image
    
    Write-Host ""
    Write-Host "${ANSI.TEXT_SECONDARY}Running container with command: docker $($args -join ' ')${ANSI.RESET}"
    
    try {
        $result = & docker $args 2>&1
        
        if ($LASTEXITCODE -eq 0) {
            Write-Host "${ANSI.GREEN}âœ“ Container started successfully${ANSI.RESET}"
            if ($result) {
                Write-Host "${ANSI.TEXT_SECONDARY}Output: $result${ANSI.RESET}"
            }
        } else {
            Write-Host "${ANSI.RED}Ã— Failed to start container${ANSI.RESET}"
            if ($result) {
                Write-Host "${ANSI.TEXT_SECONDARY}Error: $result${ANSI.RESET}"
            }
        }
    } catch {
        Write-Host "${ANSI.RED}Ã— Error running container: $($_.Exception.Message)${ANSI.RESET}"
    }
}

function Show-DockerStats {
    Write-Host ""
    Write-Host "${ANSI.TEXT_SECONDARY}Retrieving Docker container statistics...${ANSI.RESET}"
    
    try {
        Write-Host ""
        Write-Host "${ANSI.PURPLE}${ANSI.BOLD}CONTAINER STATISTICS${ANSI.RESET}"
        Write-Host "${ANSI.BORDER}â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€${ANSI.RESET}"
        
        # Run stats for 5 seconds
        $process = Start-Process -FilePath "docker" -ArgumentList "stats", "--no-stream", "--format", "table {{.Container}}\t{{.Name}}\t{{.CPUPerc}}\t{{.MemUsage}}\t{{.NetIO}}\t{{.BlockIO}}" -NoNewWindow -PassThru -RedirectStandardOutput "$env:TEMP\docker_stats.out" -RedirectStandardError "$env:TEMP\docker_stats.err"
        
        # Wait for process to complete
        $process.WaitForExit(5000)
        
        if ($process.ExitCode -eq 0) {
            $output = Get-Content "$env:TEMP\docker_stats.out" -ErrorAction SilentlyContinue
            if ($output) {
                Write-Host ($output -join "`n")
            } else {
                Write-Host "${ANSI.TEXT_SECONDARY}No containers running${ANSI.RESET}"
            }
        } else {
            $errorOutput = Get-Content "$env:TEMP\docker_stats.err" -ErrorAction SilentlyContinue
            if ($errorOutput) {
                Write-Host "${ANSI.RED}Ã— $($errorOutput)${ANSI.RESET}"
            } else {
                Write-Host "${ANSI.RED}Ã— Error retrieving stats${ANSI.RESET}"
            }
        }
        
        # Clean up temp files
        Remove-Item "$env:TEMP\docker_stats.out" -ErrorAction SilentlyContinue
        Remove-Item "$env:TEMP\docker_stats.err" -ErrorAction SilentlyContinue
    } catch {
        Write-Host "${ANSI.RED}Ã— Error retrieving stats: $($_.Exception.Message)${ANSI.RESET}"
    }
}
