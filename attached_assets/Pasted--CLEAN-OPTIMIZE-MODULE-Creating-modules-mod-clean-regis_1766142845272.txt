function Clean-TempFiles {
    Write-Host ""
    Write-Host "${ANSI.PURPLE}${ANSI.BOLD}TEMPORARY FILES CLEANER${ANSI.RESET}"
    Write-Host "${ANSI.BORDER}‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ${ANSI.RESET}"
    Write-Host ""
    
    Write-Host "${ANSI.TEXT_SECONDARY}Scanning for temporary files...${ANSI.RESET}"
    
    $tempPaths = @(
        "$env:TEMP",
        "$env:SystemRoot\Temp",
        "$env:LOCALAPPDATA\Temp"
    )
    
    $totalFreed = 0
    
    foreach ($path in $tempPaths) {
        if (Test-Path $path) {
            Write-Host "${ANSI.TEXT_SECONDARY}Scanning $path...${ANSI.RESET}"
            
            try {
                $beforeSize = (Get-ChildItem -Path $path -Recurse -Force -ErrorAction SilentlyContinue | Measure-Object -Property Length -Sum).Sum
                Get-ChildItem -Path "$path\*" -Recurse -Force -ErrorAction SilentlyContinue | Remove-Item -Force -Recurse -ErrorAction SilentlyContinue
                $afterSize = (Get-ChildItem -Path $path -Recurse -Force -ErrorAction SilentlyContinue | Measure-Object -Property Length -Sum).Sum
                $freed = $beforeSize - $afterSize
                $totalFreed += $freed
                
                Write-Host " ${ANSI.GREEN}‚úì Freed $(Format-KnouxBytes $freed)${ANSI.RESET}"
            } catch {
                Write-Host " ${ANSI.RED}√ó Error cleaning $path${ANSI.RESET}"
            }
        }
    }
    
    Write-Host ""
    Write-Host "${ANSI.GREEN}‚úì Total freed: $(Format-KnouxBytes $totalFreed)${ANSI.RESET}"
}

function List-StartupPrograms {
    Write-Host ""
    Write-Host "${ANSI.TEXT_SECONDARY}Retrieving startup programs...${ANSI.RESET}"
    
    # Registry startup locations
    $startupLocations = @(
        "HKCU:\Software\Microsoft\Windows\CurrentVersion\Run"
        "HKCU:\Software\Microsoft\Windows\CurrentVersion\RunOnce"
        "HKLM:\Software\Microsoft\Windows\CurrentVersion\Run"
        "HKLM:\Software\Microsoft\Windows\CurrentVersion\RunOnce"
    )
    
    $startupPrograms = @()
    
    foreach ($location in $startupLocations) {
        if (Test-Path $location) {
            try {
                $items = Get-ItemProperty -Path $location -ErrorAction SilentlyContinue
                $properties = $items.PSObject.Properties | Where-Object { $_.Name -notin @("PSPath", "PSParentPath", "PSChildName", "PSDrive", "PSProvider") }
                
                foreach ($prop in $properties) {
                    $startupPrograms += [PSCustomObject]@{
                        Name = $prop.Name
                        Command = $prop.Value
                        Location = $location
                        Enabled = $true
                    }
                }
            } catch {
                Write-Host "${ANSI.RED}√ó Error accessing $location${ANSI.RESET}"
            }
        }
    }
    
    # Startup folder
    $startupFolder = "$env:APPDATA\Microsoft\Windows\Start Menu\Programs\Startup"
    if (Test-Path $startupFolder) {
        $items = Get-ChildItem -Path $startupFolder -File -ErrorAction SilentlyContinue
        foreach ($item in $items) {
            $startupPrograms += [PSCustomObject]@{
                Name = $item.Name
                Command = $item.FullName
                Location = "Startup Folder"
                Enabled = $true
            }
        }
    }
    
    if ($startupPrograms.Count -eq 0) {
        Write-Host "${ANSI.TEXT_SECONDARY}No startup programs found${ANSI.RESET}"
        return
    }
    
    Write-Host ""
    Write-Host "${ANSI.PURPLE}${ANSI.BOLD}STARTUP PROGRAMS ($($startupPrograms.Count))${ANSI.RESET}"
    Write-Host "${ANSI.BORDER}‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ${ANSI.RESET}"
    
    $startupPrograms | ForEach-Object {
        Write-Host " ${ANSI.TEXT_PRIMARY}$($_.Name)${ANSI.RESET}"
        Write-Host "   ${ANSI.TEXT_SECONDARY}Command:${ANSI.RESET} ${ANSI.TEXT_PRIMARY}$($_.Command)${ANSI.RESET}"
        Write-Host "   ${ANSI.TEXT_SECONDARY}Location:${ANSI.RESET} ${ANSI.TEXT_PRIMARY}$($_.Location)${ANSI.RESET}"
        Write-Host ""
    }
}

function Add-StartupProgram {
    Write-Host ""
    Write-Host "${ANSI.TEXT_SECONDARY}Adding programs to startup requires administrator privileges for HKLM${ANSI.RESET}"
    Write-Host "${ANSI.TEXT_SECONDARY}For HKCU (recommended), no special privileges are needed${ANSI.RESET}"
    
    Write-Host ""
    Write-Host "${ANSI.TEXT_SECONDARY}Enter program name:${ANSI.RESET}"
    Write-Host "${ANSI.PURPLE}>>${ANSI.RESET} " -NoNewline
    $programName = Read-Host
    
    Write-Host ""
    Write-Host "${ANSI.TEXT_SECONDARY}Enter program path:${ANSI.RESET}"
    Write-Host "${ANSI.PURPLE}>>${ANSI.RESET} " -NoNewline
    $programPath = Read-Host
    
    if ([string]::IsNullOrWhiteSpace($programName) -or [string]::IsNullOrWhiteSpace($programPath)) {
        Write-Host "${ANSI.RED}√ó Name and path cannot be empty${ANSI.RESET}"
        return
    }
    
    if (-not (Test-Path $programPath)) {
        Write-Host "${ANSI.RED}√ó Program not found: $programPath${ANSI.RESET}"
        return
    }
    
    Write-Host ""
    Write-Host "${ANSI.TEXT_SECONDARY}Select location:${ANSI.RESET}"
    Write-Host " ${ANSI.PURPLE}1${ANSI.RESET} ${ANSI.TEXT_PRIMARY}Current User (HKCU)${ANSI.RESET}"
    Write-Host " ${ANSI.PURPLE}2${ANSI.RESET} ${ANSI.TEXT_PRIMARY}All Users (HKLM - requires admin)${ANSI.RESET}"
    Write-Host " ${ANSI.PURPLE}3${ANSI.RESET} ${ANSI.TEXT_PRIMARY}Startup Folder${ANSI.RESET}"
    Write-Host ""
    
    $locationChoice = Read-ValidatedSubInput -Max 3
    
    try {
        switch ($locationChoice) {
            1 {
                $regPath = "HKCU:\Software\Microsoft\Windows\CurrentVersion\Run"
                Set-ItemProperty -Path $regPath -Name $programName -Value $programPath
                Write-Host "${ANSI.GREEN}‚úì Added to current user startup${ANSI.RESET}"
            }
            2 {
                if (-not (Test-AdminPrivilege)) {
                    Write-Host "${ANSI.ORANGE}‚ö† Administrator privileges required${ANSI.RESET}"
                    return
                }
                $regPath = "HKLM:\Software\Microsoft\Windows\CurrentVersion\Run"
                Set-ItemProperty -Path $regPath -Name $programName -Value $programPath
                Write-Host "${ANSI.GREEN}‚úì Added to all users startup${ANSI.RESET}"
            }
            3 {
                $startupFolder = "$env:APPDATA\Microsoft\Windows\Start Menu\Programs\Startup"
                $shortcutPath = Join-Path $startupFolder "$programName.lnk"
                
                # Create shortcut (simplified approach)
                $shell = New-Object -ComObject WScript.Shell
                $shortcut = $shell.CreateShortcut($shortcutPath)
                $shortcut.TargetPath = $programPath
                $shortcut.Save()
                
                Write-Host "${ANSI.GREEN}‚úì Added to startup folder${ANSI.RESET}"
            }
        }
    } catch {
        Write-Host "${ANSI.RED}√ó Error adding startup program: $($_.Exception.Message)${ANSI.RESET}"
    }
}

function Defrag-Disk {
    Write-Host ""
    Write-Host "${ANSI.PURPLE}${ANSI.BOLD}DISK DEFRAGMENTER${ANSI.RESET}"
    Write-Host "${ANSI.BORDER}‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ${ANSI.RESET}"
    Write-Host ""
    
    Write-Host "${ANSI.TEXT_SECONDARY}Defragmenting disks requires administrator privileges${ANSI.RESET}"
    
    if (-not (Test-AdminPrivilege)) {
        Write-Host "${ANSI.ORANGE}‚ö† Administrator privileges required for defragmentation${ANSI.RESET}"
        return
    }
    
    # Get fixed drives
    $drives = Get-WmiObject -Class Win32_LogicalDisk | Where-Object { $_.DriveType -eq 3 }
    
    Write-Host ""
    Write-Host "${ANSI.TEXT_SECONDARY}Fixed drives found:${ANSI.RESET}"
    $drives | ForEach-Object {
        $freeSpace = [math]::Round(($_.FreeSpace / 1GB), 2)
        $totalSpace = [math]::Round(($_.Size / 1GB), 2)
        $usedPercent = [math]::Round(((($_.Size - $_.FreeSpace) / $_.Size) * 100), 2)
        
        Write-Host " ${ANSI.TEXT_PRIMARY}$($_.DeviceID)${ANSI.RESET} ${ANSI.TEXT_SECONDARY}($usedPercent% full, $freeSpace GB free of $totalSpace GB)${ANSI.RESET}"
    }
    
    Write-Host ""
    Write-Host "${ANSI.TEXT_SECONDARY}Select optimization type:${ANSI.RESET}"
    Write-Host " ${ANSI.PURPLE}1${ANSI.RESET} ${ANSI.TEXT_PRIMARY}Analyze drives${ANSI.RESET}"
    Write-Host " ${ANSI.PURPLE}2${ANSI.RESET} ${ANSI.TEXT_PRIMARY}Defragment drives${ANSI.RESET}"
    Write-Host " ${ANSI.PURPLE}3${ANSI.RESET} ${ANSI.TEXT_PRIMARY}Optimize drives (Windows 8+)${ANSI.RESET}"
    Write-Host ""
    
    $optChoice = Read-ValidatedSubInput -Max 3
    
    switch ($optChoice) {
        1 {
            Write-Host ""
            Write-Host "${ANSI.TEXT_SECONDARY}Analyzing drives...${ANSI.RESET}"
            $drives | ForEach-Object {
                Write-Host "${ANSI.TEXT_SECONDARY}Analyzing $($_.DeviceID)...${ANSI.RESET}"
                # In real implementation: defrag.exe $($_.DeviceID) /A /U
                Write-Host " ${ANSI.TEXT_SECONDARY}Simulation: Analysis would be performed here${ANSI.RESET}"
            }
        }
        2 {
            Write-Host ""
            Write-Host "${ANSI.ORANGE}? Traditional defragmentation requires significant time${ANSI.RESET}"
            $confirm = Confirm-KnouxAction "Continue with defragmentation?" "N"
            
            if ($confirm) {
                $drives | ForEach-Object {
                    Write-Host "${ANSI.TEXT_SECONDARY}Defragmenting $($_.DeviceID)...${ANSI.RESET}"
                    # In real implementation: defrag.exe $($_.DeviceID) /U
                    Write-Host " ${ANSI.TEXT_SECONDARY}Simulation: Defragmentation would be performed here${ANSI.RESET}"
                    Start-Sleep -Seconds 2
                    Write-Host " ${ANSI.GREEN}‚úì Completed defragmentation of $($_.DeviceID)${ANSI.RESET}"
                }
                Write-Host ""
                Write-Host "${ANSI.GREEN}‚úì Defragmentation process completed${ANSI.RESET}"
            }
        }
        3 {
            Write-Host ""
            Write-Host "${ANSI.TEXT_SECONDARY}Using Windows built-in optimizer...${ANSI.RESET}"
            $drives | ForEach-Object {
                Write-Host "${ANSI.TEXT_SECONDARY}Optimizing $($_.DeviceID)...${ANSI.RESET}"
                # In real implementation: Optimize-Volume -DriveLetter $($_.DeviceID -replace ':','') -Defrag -Verbose
                Write-Host " ${ANSI.TEXT_SECONDARY}Simulation: Optimization would be performed here${ANSI.RESET}"
                Start-Sleep -Seconds 1
                Write-Host " ${ANSI.GREEN}‚úì Completed optimization of $($_.DeviceID)${ANSI.RESET}"
            }
            Write-Host ""
            Write-Host "${ANSI.GREEN}‚úì Disk optimization completed${ANSI.RESET}"
        }
    }
}
